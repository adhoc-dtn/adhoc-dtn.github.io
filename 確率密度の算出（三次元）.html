In [130]:
#多次元分布では分布間の関係性を共分散行列という形で作る必要がある。それを作るやつ
cov_mat = np.array([])
d1=0
while d1 < dim:
    d2=0
    line=np.array([])
    #行を作る
    while d2 < dim:
        line=np.append(line, rho_ar[d1][d2]*sig_ar[d1]*sig_ar[d2])
        d2+=1
    if(d1==0):#行を作る
        cov_mat=line
    else:#行列を連結する
        cov_mat=np.vstack([cov_mat, line])

    d1+=1
