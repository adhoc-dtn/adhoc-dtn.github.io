#多次元同時正規分布を作る
seed=1
num_rand_list=[500,1000,3000]
split = 5
method = "kde"
index=0
for mu_mat, cov_mat,dim in zip(mu_mat_array,cov_mat_array,dim_list):
    for num_rand in num_rand_list:
        #多次元正規分布の乱数生成
        multi_rand_tuple = gm.generate_multivariate_normal_upper3d(mu_mat, cov_mat, num_rand, seed)
        #カーネル密度推定
        kde              = gm2.est_kde(multi_rand_tuple)
        #多次元正規分布の乱数生成
        multi_rand_tuple_test = gm.generate_multivariate_normal_upper3d(mu_mat, cov_mat, num_rand, seed*2)
        #尤度計算（カーネル密度推定）
        est_likelihood     = gm2.calc_KDE_log_likelihood(kde,multi_rand_tuple_test)
        #尤度計算（元の分布に基づく）
        orig_likelihood    = gm2.calc_MLE_log_likelihood(multi_rand_tuple_test, mu_mat, cov_mat,)
        #相関算出
        rho = np.corrcoef(est_likelihood, orig_likelihood)[0,1]
        #スペアマン相関算出
        sp_rho, sp_pval = stats.spearmanr(est_likelihood, orig_likelihood)
        
        df_data = pd.DataFrame(data=[[method,dim,num_rand,rho,sp_rho,sp_pval,sum(est_likelihood),sum(orig_likelihood)]],
                          columns=["method","dim","num_rand","rho","sp_rho","sp_pval","est_log_likelihood","orig_log_likelihood"])
        if(index==0):
            df_con = df_data
        else:
            df_con = pd.concat([df_con,df_data])

        index+=1
